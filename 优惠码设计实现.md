## 🧩 一、优惠码模型设计（`app/models/coupon.py`）

```python
from sqlalchemy import Column, Integer, String, Boolean, DateTime, Text, ForeignKey
from sqlalchemy.orm import relationship
from app.database import Base
from datetime import datetime
from typing import Optional

class Coupon(Base):
    __tablename__ = "coupons"

    id = Column(Integer, primary_key=True, index=True)
    code = Column(String(64), unique=True, index=True, nullable=False)
    
    # 渠道
    channel = Column(String(32), nullable=False, default="default")  # 闲鱼、抖音、淘宝等

    # 适用范围：all / category:101 / product:2001
    scope = Column(String(64), nullable=False, default="all")  # all, category:1, product:100

    # 最大使用次数
    max_uses = Column(Integer, nullable=False, default=1)

    # 已使用次数
    uses_count = Column(Integer, nullable=False, default=0)

    # 是否绑定特定用户（若绑定，则只能该用户使用，且 max_uses = 1）
    bound_user_id = Column(Integer, nullable=True, index=True)

    # 锁定状态（分发时锁定，防止并发使用）
    is_locked = Column(Boolean, default=False, nullable=False)

    # 是否有效
    is_active = Column(Boolean, default=True, nullable=False)

    # 有效期类型
    expire_type = Column(String(32), nullable=False, default="permanent")
    expire_start = Column(DateTime, nullable=True)
    expire_end = Column(DateTime, nullable=True)

    # 关联规则编号（外键引用规则集）
    rule_id = Column(Integer, nullable=False)

    # 创建时间、更新时间
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

    # 关系：关联规则集（逻辑外键）
    rule = relationship("CouponRule", back_populates="coupons")

    # 用于判断是否过期
    def is_expired(self) -> bool:
        if self.expire_type == "permanent":
            return False
        if not self.expire_end:
            return True
        return datetime.utcnow() > self.expire_end

    # 用于判断是否已绑定用户
    def is_bound_to_user(self, user_id: int) -> bool:
        return self.bound_user_id is not None and self.bound_user_id == user_id

    # 用于判断是否可使用
    def can_use(self, user_id: int, product_id: Optional[int] = None, category_id: Optional[int] = None) -> bool:
        # 1. 是否有效
        if not self.is_active:
            return False
        # 2. 是否过期
        if self.is_expired():
            return False
        # 3. 是否被锁定
        if self.is_locked:
            return False
        # 4. 是否已达到最大使用次数
        if self.uses_count >= self.max_uses:
            return False
        # 5. 是否绑定用户？且是否是该用户？
        if self.bound_user_id is not None and self.bound_user_id != user_id:
            return False
        # 6. 适用范围判断（all 表示不限制）
        if self.scope == "all":
            return True
        if self.scope.startswith("category:"):
            cat_id = int(self.scope.split(":")[1])
            return category_id == cat_id
        if self.scope.startswith("product:"):
            prod_id = int(self.scope.split(":")[1])
            return product_id == prod_id
        return False
```

## 🛠️ 三、优惠码使用服务（`app/services/coupon_service.py`）

```python
from typing import Optional, Dict, Any
from fastapi import HTTPException, Depends
from sqlalchemy.orm import Session
from app.models.coupon import Coupon
from app.models.coupon_usage import CouponUsage
from app.cache.redis_client import get_redis_client
from app.events.kafka_events import publish_coupon_used_event
from app.database import get_db
from datetime import datetime
import logging

logger = logging.getLogger(__name__)

class CouponService:
    def __init__(self, db: Session):
        self.db = db

    def generate_coupon_code(self) -> str:
        import random
        import string
        return ''.join(random.choices(string.ascii_uppercase + string.digits, k=8))

    def create_coupon(self, data: Dict[str, Any], admin_id: int) -> Coupon:
        rule = self.db.query(CouponRule).filter(CouponRule.id == data["rule_id"]).first()
        if not rule:
            raise HTTPException(status_code=404, detail="Rule not found")

        # 只允许超管创建
        if not self.is_superadmin(admin_id):
            raise HTTPException(status_code=403, detail="Only super admin can create coupons")

        # 处理有效期
        expire_start = data.get("expire_start")
        expire_end = data.get("expire_end")

        if data["expire_type"] == "custom":
            if not expire_start or not expire_end:
                raise HTTPException(status_code=400, detail="Custom expire range requires start and end")
        elif data["expire_type"] == "permanent":
            expire_start = None
            expire_end = None
        else:
            # 一周、一个月等
            from datetime import timedelta
            now = datetime.utcnow()
            if data["expire_type"] == "week":
                expire_end = now + timedelta(weeks=1)
            elif data["expire_type"] == "month":
                expire_end = now + timedelta(days=30)
            elif data["expire_type"] == "quarter":
                expire_end = now + timedelta(days=90)
            else:
                expire_end = None

        code = self.generate_coupon_code()

        coupon = Coupon(
            code=code,
            channel=data["channel"],
            scope=data["scope"],
            max_uses=data["max_uses"],
            bound_user_id=data.get("bound_user_id") if data.get("bound_user_id") else None,
            is_locked=data.get("is_locked", False),
            is_active=data.get("is_active", True),
            expire_type=data["expire_type"],
            expire_start=expire_start,
            expire_end=expire_end,
            rule_id=data["rule_id"]
        )

        self.db.add(coupon)
        self.db.commit()
        self.db.refresh(coupon)
        logger.info(f"Admin {admin_id} created coupon: {code}")

        return coupon

    def use_coupon(self, code: str, user_id: int, order_id: str, product_id: int = None, category_id: int = None) -> Dict[str, Any]:
        # 1. 从 Redis 限流（防止刷）
        redis_client = get_redis_client()
        key = f"coupon:use:rate_limit:{code}:{user_id}"
        if redis_client.get(key):
            raise HTTPException(status_code=429, detail="Too many requests")

        # 2. 查询优惠码
        coupon = self.db.query(Coupon).filter(Coupon.code == code).first()
        if not coupon:
            raise HTTPException(status_code=404, detail="Coupon not found")

        # 3. 校验可用性
        if not coupon.can_use(user_id=user_id, product_id=product_id, category_id=category_id):
            raise HTTPException(status_code=400, detail="Coupon cannot be used")

        # 4. 锁定优惠码（防止并发使用）
        if coupon.is_locked:
            raise HTTPException(status_code=400, detail="Coupon is locked")

        # 5. 更新使用次数
        coupon.uses_count += 1
        self.db.commit()

        # 6. 记录使用
        usage = CouponUsage(
            coupon_id=coupon.id,
            user_id=user_id,
            order_id=order_id,
            used_at=datetime.utcnow(),
            product_id=product_id,
            category_id=category_id,
            discount_type=coupon.rule.discount_type,
            discount_value=coupon.rule.discount_value
        )
        self.db.add(usage)
        self.db.commit()

        # 7. 发送 Kafka 事件（异步处理通知、统计等）
        publish_coupon_used_event(
            coupon_code=code,
            user_id=user_id,
            order_id=order_id,
            amount=coupon.rule.discount_value,
            discount_type=coupon.rule.discount_type
        )

        # 8. 记录日志
        logger.info(f"User {user_id} used coupon {code} for order {order_id}")

        return {
            "code": coupon.code,
            "discount_amount": coupon.rule.discount_value,
            "discount_type": coupon.rule.discount_type,
            "used_at": usage.used_at
        }

    def is_superadmin(self, user_id: int) -> bool:
        # 实际项目中应从数据库或缓存查
        return user_id == 1  # 示例：超管 ID 为 1
```

## 📤 四、Kafka 事件生产（`app/events/kafka_events.py`）

```python
from kafka import KafkaProducer
import json
import os

producer = KafkaProducer(
    bootstrap_servers=os.getenv("KAFKA_BOOTSTRAP_SERVERS", "localhost:9092")),
    value_serializer=lambda v: json.dumps(v).encode('utf-8')
)

def publish_coupon_used_event(**data):
    topic = "coupon.used"
    producer.send(topic, data)
    producer.flush()
```

from sqlalchemy import Column, Integer, String, DateTime, ForeignKey
